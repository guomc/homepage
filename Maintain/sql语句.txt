进入数据库
mysql -u root -p

建立表空间
create database guomcdb character set utf8;
use guomcdb;

建表，不知道RSContent到底多大，varchar(大值)相当于text
create table ResInfo(
RSID int primary key auto_increment,
RSName varchar(60),
RSContent LONGTEXT,
RSReleaseDate datetime);
