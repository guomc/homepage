使用sqlite3数据库

头文件是qdebug.h、QtSql/QSqlDatabase、QtSql/QSqlError、QtSql/QSqlQuery

1.在pro文件添加语句：QT += sql

2.执行语句

QSqlDatabase database=QSqlDatabase::addDatabase("QSQLITE");

database.setDatabaseName("MyDatabase.db");

if(!database.open())

       qDebug()<<"连接数据库错误："<<database.lastError();

else

       qDebug()<<"连接数据库成功！";

//执行无返回值的SQL语句

QSqlQuery sql_query;

if(!sql_query.exec("create table students(id int primary key,name text,age int);"))

        qDebug()<<"建表失败："<<sql_query.lastError();

else

        qDebug()<<"建表成功！";

//执行有返回值的SQL语句

if(!sql_query.exec("select * from students;"))

        qDebug()<<"查失败："<<sql_query.lastError();

else

        while(sql_query.next())

        {

            int id=sql_query.value(0).toInt();

            QString name=sql_query.value(1).toString();

            int age=sql_query.value(2).toInt();

            qDebug()<<QString("id:%1---name:%2---age:%3").arg(id).arg(name).arg(age);

        }

database.close();

与php交互

头文件是QNetworkReply、QNetworkAccessManager、QNetworkReply

//头文件中定义全局变量

QNetworkReply *reply;

QNetworkAccessManager *manager;

//初始化中绑定信号和槽，SIGNAL和SLOT中填函数的表示，当信号发生时也会触发槽

manager = new QNetworkAccessManager;

connect(manager,SIGNAL(finished(QNetworkReply*)),this,SLOT(panduan(QNetworkReply*)));

//点击界面上的一些元素，触发事件

QString str = "zhanghao="+ui->textEdit_3->toPlainText()+"&mima="+ui->textEdit_4->toPlainText();

QString url = "http://localhost/guomc/php/sy.php";

QByteArray *parmanager = new QByteArray(str.toUtf8());

QNetworkRequest *request = new QNetworkRequest(QUrl(url));

request->setHeader(QNetworkRequest::ContentTypeHeader,QVariant("application/x-www-form-urlencoded"));

reply = manager->post(*request,*parmanager);

//处理函数panduan()

QString results = "";

if(reply->error() == QNetworkReply::NoError)

{

        QByteArray bytes = reply->readAll();

        results.prepend(bytes);

        if(results.length() != 0)

        {

            if(results == "OK")

                QMessageBox::information(0,"提示","登陆成功！");

            else

                QMessageBox::information(0,"提示","账号或者密码错误，请重新输入！");

       }

}
